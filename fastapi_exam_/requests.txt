     -H "Content-Type: application/json" \
     -d '{"userid": 1, "name": "Alice", "subscription": "premium"}'

     -H "Content-Type: application/json" \
     -d '{"userid": 4, "name": "Sabine", "subscription": "modification contrat"}'
